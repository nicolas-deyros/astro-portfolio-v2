<script>
	document.addEventListener('astro:page-load', () => {
		const mobileMenuButton = document.getElementById('mobile-menu-button')
		const mobileMenu = document.getElementById('mobile-menu')
		const mobileMenuOverlay = document.getElementById('mobile-menu-overlay')
		const mobileMenuClose = document.getElementById('mobile-menu-close')
		const hamburgerIcon = document.getElementById('hamburger-icon')
		const closeIcon = document.getElementById('close-icon')
		const mobileNavLinks = document.querySelectorAll('.mobile-nav-link')
 
		if (!mobileMenuButton || !mobileMenu || !mobileMenuOverlay) return
 
		function openMobileMenu(): void {
			if (mobileMenu) {
				mobileMenu.classList.remove('translate-x-full')
			}
			if (mobileMenuOverlay) {
				mobileMenuOverlay.classList.remove('opacity-0', 'invisible')
			}
			hamburgerIcon?.classList.add('hidden')
			closeIcon?.classList.remove('hidden')
			mobileMenuButton?.setAttribute('aria-expanded', 'true')
			document.body.classList.add('overflow-hidden')
		}
 
		function closeMobileMenu(): void {
			if (mobileMenu) {
				mobileMenu.classList.add('translate-x-full')
			}
			if (mobileMenuOverlay) {
				mobileMenuOverlay.classList.add('opacity-0', 'invisible')
			}
			hamburgerIcon?.classList.remove('hidden')
			closeIcon?.classList.add('hidden')
			mobileMenuButton?.setAttribute('aria-expanded', 'false')
			document.body.classList.remove('overflow-hidden')
		}
 
		// Toggle mobile menu
		mobileMenuButton.addEventListener('click', () => {
			const isOpen = !mobileMenu.classList.contains('translate-x-full')
			if (isOpen) {
				closeMobileMenu()
			} else {
				openMobileMenu()
			}
		})
 
		// Close menu when clicking overlay
		mobileMenuOverlay.addEventListener('click', closeMobileMenu)
 
		// Close menu when clicking close button
		mobileMenuClose?.addEventListener('click', closeMobileMenu)
 
		// Close menu when clicking nav links
		mobileNavLinks.forEach(link => {
			link.addEventListener('click', closeMobileMenu)
		})
 
		// Close menu when clicking admin nav links
		const adminNavLinks = document.querySelectorAll(
			'.admin-menu-item[data-navlink]',
		)
		adminNavLinks.forEach(link => {
			link.addEventListener('click', closeMobileMenu)
		})
 
		// Close menu on escape key
		document.addEventListener('keydown', e => {
			if (
				e.key === 'Escape' &&
				!mobileMenu.classList.contains('translate-x-full')
			) {
				closeMobileMenu()
			}
		})
 
		// Close menu on window resize to desktop size
		window.addEventListener('resize', () => {
			if (window.innerWidth >= 1024) {
				closeMobileMenu()
			}
		})
 
		// Theme toggle functionality - handle both desktop and mobile
		const themeToggleCheckbox = document.getElementById(
			'theme-toggle-checkbox',
		) as HTMLInputElement
		const mobileThemeToggleCheckbox = document.getElementById(
			'mobile-theme-toggle-checkbox',
		) as HTMLInputElement
		const htmlElement = document.documentElement
 
		// Sync checkbox states with current theme
		function syncCheckboxStates(): void {
			const isDark = htmlElement.classList.contains('dark')
			if (themeToggleCheckbox) {
				themeToggleCheckbox.checked = isDark
			}
			if (mobileThemeToggleCheckbox) {
				mobileThemeToggleCheckbox.checked = isDark
			}
		}
 
		// Sync checkbox states on load (theme is already set by layout)
		syncCheckboxStates()
 
		// Handle desktop toggle
		if (themeToggleCheckbox) {
			themeToggleCheckbox.addEventListener('change', () => {
				if (themeToggleCheckbox.checked) {
					htmlElement.classList.add('dark')
					localStorage.setItem('theme', 'dark')
				} else {
					htmlElement.classList.remove('dark')
					localStorage.setItem('theme', 'light')
				}
				// Sync the other checkbox
				if (mobileThemeToggleCheckbox) {
					mobileThemeToggleCheckbox.checked = themeToggleCheckbox.checked
				}
			})
		}
 
		// Handle mobile toggle
		if (mobileThemeToggleCheckbox) {
			mobileThemeToggleCheckbox.addEventListener('change', () => {
				if (mobileThemeToggleCheckbox.checked) {
					htmlElement.classList.add('dark')
					localStorage.setItem('theme', 'dark')
				} else {
					htmlElement.classList.remove('dark')
					localStorage.setItem('theme', 'light')
				}
				// Sync the other checkbox
				if (themeToggleCheckbox) {
					themeToggleCheckbox.checked = mobileThemeToggleCheckbox.checked
				}
			})
		}
 
		// Admin logout functionality
		const adminLogoutBtn = document.getElementById('admin-logout-btn')
		const mobileAdminLogoutBtn = document.getElementById(
			'mobile-admin-logout-btn',
		)
 
		function checkAdminStatus(): void {
			const isAdmin = localStorage.getItem('admin_authenticated') === 'true'
			const currentIsAdminPage = window.location.pathname.startsWith('/admin')
 
			// Only proceed if we're on an admin page (elements exist)
			if (!currentIsAdminPage) return
 
			// Show/hide admin menu items - only show if authenticated
			const adminMenuItems = document.querySelectorAll('.admin-menu-item')
			const adminLogoutBtns = document.querySelectorAll(
				'#admin-logout-btn, #mobile-admin-logout-btn',
			)
 
			if (isAdmin) {
				// User is authenticated - show admin items
				adminMenuItems.forEach(item => {
					item.classList.remove('hidden')
					// For mobile menu items (links), add flex to show them properly
					if (item.tagName.toLowerCase() === 'a') {
						item.classList.add('flex')
					}
					// For desktop menu items (NavList components), restore the flex classes
					if (item.tagName.toLowerCase() === 'li') {
						item.classList.add(
							'flex',
							'list-none',
							'items-center',
							'justify-center',
						)
					}
				})
				adminLogoutBtns.forEach(btn => btn.classList.remove('hidden'))
			} else {
				// User is not authenticated - hide admin items
				adminMenuItems.forEach(item => {
					item.classList.add('hidden')
					// Remove all display classes
					item.classList.remove(
						'flex',
						'list-none',
						'items-center',
						'justify-center',
					)
				})
				adminLogoutBtns.forEach(btn => btn.classList.add('hidden'))
			}
		}
 
		async function handleLogout(): Promise<void> {
			try {
				// Call server logout to invalidate session
				await fetch('/api/auth.json', {
					method: 'POST',
					headers: { 'Content-Type': 'application/json' },
					body: JSON.stringify({ action: 'logout' }),
					credentials: 'include',
				})
			} catch (error) {
				console.error('Server logout error:', error)
			} finally {
				// Always clear local storage
				localStorage.removeItem('admin_authenticated')
				localStorage.removeItem('auth_token')
				localStorage.removeItem('device_fingerprint')
				localStorage.removeItem('session_expires')
				// Always redirect to admin login page after logout
				window.location.href = '/admin'
			}
		}
 
		// Check admin status on page load
		checkAdminStatus()
 
		// Handle logout button clicks
		adminLogoutBtn?.addEventListener('click', handleLogout)
		mobileAdminLogoutBtn?.addEventListener('click', handleLogout)
 
		// Listen for storage changes (when user logs in/out in another tab)
		window.addEventListener('storage', e => {
			if (e.key === 'admin_authenticated') {
				checkAdminStatus()
			}
		})
 
		// Check admin status when navigating between pages
		document.addEventListener('astro:after-swap', checkAdminStatus)
	})
</script>
